🚀 Inicio Rápido

Crea un viaje resiliente en segundos:

from wpipe import Pipeline, Parallel, step

@step(name="Tarea", retry_count=3)
def mi_tarea(ctx):
    return {"status": "ok"}

# Orquestación con reanudación y paralelismo
viaje = Pipeline(pipeline_name="Viaje_Pro")
viaje.add_checkpoint("hito_1", expression="status == 'ok'")

viaje.set_steps([
    mi_tarea,
    Parallel(steps=[paso_A, paso_B], max_workers=2)
])

viaje.run({"inicio": True})
